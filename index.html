1. Form Validation

function validateForm(name, quantity) {
  const nameError = name.trim() === "" ? "Product name is required" : "";
  const quantityError = quantity.trim() === "" ? "Quantity is required" : "";
  return { nameError, quantityError, isValid: nameError === "" && quantityError === "" };
}

2. Debounce / Throttle

function debounce(fn, delay) {
  let timer;
  return (...args) => { clearTimeout(timer); timer = setTimeout(() => fn(...args), delay); };
}

5. useOutsideClick

function useOutsideClick(ref, onOutsideClick) {
  useEffect(() => {
    const handle = (e) => {
      if (ref.current && !ref.current.contains(e.target)) onOutsideClick();
    };
    document.addEventListener("click", handle);
    return () => document.removeEventListener("click", handle);
  }, [ref, onOutsideClick]);
}
